//levantar y abrir el proyecto en el navegador
ng serve -o

//Crear modulo nuevo por consola.
ng g m [ruta del modulo]

//Crear componente nuevo por consola.
ng g c [ruta del componente] --skip-tests //--skip-tests evita crear el archivo de pruebas
ej: ng g c dbz/mainPage --skipTests

//crear build de prod
ng build --prod
